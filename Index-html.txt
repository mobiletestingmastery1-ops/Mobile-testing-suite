        // Set your access code here (change this to your unique code)
        const VALID_ACCESS_CODE = "MTM2024";

        function checkAccess() {
            const inputCode = document.getElementById('accessCode').value;
            const errorMessage = document.getElementById('errorMessage');
            
            if (inputCode === VALID_ACCESS_CODE) {
                // Grant access
                document.getElementById('accessGate').style.display = 'none';
                document.getElementById('mainContent').style.display = 'block';
                
                // Store access in session
                sessionStorage.setItem('authenticated', 'true');
                
                // Clear the input
                document.getElementById('accessCode').value = '';
                errorMessage.style.display = 'none';
            } else {
                // Show error
                errorMessage.style.display = 'block';
                document.getElementById('accessCode').value = '';
            }
        }

        function logout() {
            if (confirm('Are you sure you want to logout?')) {
                sessionStorage.removeItem('authenticated');
                document.getElementById('accessGate').style.display = 'block';
                document.getElementById('mainContent').style.display = 'none';
                window.scrollTo(0, 0);
            }
        }

        function downloadDocument(docNumber) {
            const docNames = {
                1: 'Client Intake & Consent Form',
                2: 'Results Release & Authorization',
                3: 'Professional Results Template',
                4: 'Mobile Service Checklist',
                5: 'Service Menu & Price List',
                6: 'Client Communication Templates'
            };
            
            alert(`Downloading: ${docNames[docNumber]}\n\nIn production, this would download the actual document file.\n\nFor now, you can print each document directly from the artifact viewer.`);
        }

        // Check if user is already authenticated
        window.onload = function() {
            if (sessionStorage.getItem('authenticated') === 'true') {
                document.getElementById('accessGate').style.display = 'none';
                document.getElementById('mainContent').style.display = 'block';
            }
        };
